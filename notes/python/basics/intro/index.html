    Comments

    Single line comment

    #This is a comment
print("Hello, World!")

    Inline comment

    print("Hello, World!") #This is a comment

    Multi-line comments

    #This is a comment
#written in
#more than just one line
print("Hello, World!")

    or

    """
This is a comment
written in
more than just one line
"""
